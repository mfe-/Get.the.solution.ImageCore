public byte[] Resize(byte[] originalImage, int newWidth, int newHeight)
        {
            var outputStream = new MemoryStream();

            IImageFormat format;
            using (Image<Rgba32> image = Image.Load(originalImage, out format))
            {
                var t = image.Clone(x => x
                    .Resize(newWidth, newHeight)
                     );

                t.Save(outputStream, format);
            }

            return outputStream.ToArray();
        }
